<!-- FullCalendar 6 global bundle init (from RESEARCH.md Pattern 2) -->
```javascript
var calendar = new FullCalendar.Calendar( el, {
    initialView:  settings.calendarView === 'list' ? 'listMonth' : 'dayGridMonth',
    headerToolbar: {
        left:   'prev,next today',
        center: 'title',
        right:  ''  // view toggle is our own UI
    },
    events: {
        url:    settings.restUrl,
        method: 'GET',
        extraParams: { _fc: 1, per_page: 200 },
        failure: function() { /* show error state */ }
    },
    eventClick: function( info ) {
        window.location.href = info.event.url;
        info.jsEvent.preventDefault();
    }
} );
calendar.render();
```

<!-- Enqueue guard — only on events directory page -->
```php
// In bp-events-loader.php or a new enqueue function in bp-events-filters.php:
function bp_events_enqueue_calendar_assets() {
    if ( ! bp_is_current_component( 'events' ) ) {
        return;
    }
    wp_enqueue_script( 'fullcalendar', /* local URL */, array(), '6.1.20', true );
    wp_enqueue_script( 'bp-events-calendar', /* local URL */, array( 'fullcalendar' ), BP_PLATFORM_VERSION, true );
    wp_localize_script( 'bp-events-calendar', 'bpEventsSettings', array( /* see above */ ) );
    wp_enqueue_style( 'bp-events', /* local URL */, array(), BP_PLATFORM_VERSION );
}
add_action( 'wp_enqueue_scripts', 'bp_events_enqueue_calendar_assets' );
```
